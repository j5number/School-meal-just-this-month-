    <script>

        const allergyMap = {

            "1": "난류", "2": "우유", "3": "메밀", "4": "땅콩", "5": "대두", "6": "밀", "7": "고등어", "8": "게", "9": "새우", "10": "돼지고기", "11": "복숭아", "12": "토마토", "13": "아황산류", "14": "호두", "15": "닭고기", "16": "쇠고기", "17": "오징어", "18": "조개류", "19": "잣"

        };


        const state = {

            currentSchool: JSON.parse(localStorage.getItem('selectedSchool')) || null,

            favorites: JSON.parse(localStorage.getItem('favoriteSchools')) || [],

            favDishes: JSON.parse(localStorage.getItem('favDishes')) || [],

            badDishes: JSON.parse(localStorage.getItem('badDishes')) || [],

            selectedDate: new Date().toISOString().split('T')[0].replace(/-/g, '')

        };


        const ui = {

            showSection(section) {

                document.querySelectorAll('main > section').forEach(s => s.classList.add('hidden'));

                document.querySelectorAll('nav button').forEach(b => b.classList.remove('text-blue-600', 'scale-110'));

                

                if (section === 'meal') {

                    if (!state.currentSchool) return this.showSection('setup');

                    document.getElementById('meal-section').classList.remove('hidden');

                    document.getElementById('nav-meal').classList.add('text-blue-600', 'scale-110');

                    app.initMealView();

                } else if (section === 'setup') {

                    document.getElementById('setup-section').classList.remove('hidden');

                    document.getElementById('nav-setup').classList.add('text-blue-600', 'scale-110');

                } else if (section === 'favorites') {

                    document.getElementById('favorites-section').classList.remove('hidden');

                    document.getElementById('nav-fav').classList.add('text-blue-600', 'scale-110');

                    this.renderFavorites();

                }

            },

            toggleLoader(show) {

                document.getElementById('global-loader').classList.toggle('hidden', !show);

            },

            renderFavorites() {

                const list = document.getElementById('favorites-list');

                let html = '';


                // 선호/기피 음식 관리

                const renderChipList = (title, items, colorClass, icon, toggleFunc) => `

                    <div>

                        <h4 class="text-xs font-bold ${colorClass} mb-3 flex items-center gap-1">

                            <i class="${icon}"></i> ${title}

                        </h4>

                        <div class="flex flex-wrap gap-2">

                            ${items.length ? items.map(dish => `

                                <div class="bg-white border px-3 py-2 rounded-xl flex items-center gap-2 shadow-sm">

                                    <span class="text-sm font-medium text-gray-700">${dish}</span>

                                    <button onclick="app.${toggleFunc}('${dish}')" class="text-gray-300 hover:text-red-500"><i class="fa-solid fa-circle-xmark"></i></button>

                                </div>

                            `).join('') : `<p class="text-xs text-gray-400 italic">등록된 음식이 없습니다.</p>`}

                        </div>

                    </div>

                `;


                html += renderChipList('내가 좋아하는 반찬', state.favDishes, 'text-blue-500', 'fa-solid fa-star', 'toggleFavDish');

                html += renderChipList('내가 싫어하는 반찬', state.badDishes, 'text-red-500', 'fa-solid fa-circle-minus', 'toggleBadDish');


                // 즐겨찾는 학교

                html += `<div class="pt-4 border-t mt-4">

                    <h4 class="text-xs font-bold text-gray-400 mb-3 uppercase">즐겨찾는 학교</h4>

                    ${state.favorites.length ? state.favorites.map(s => `

                        <div class="bg-white p-4 rounded-2xl border mb-2 flex justify-between items-center shadow-sm">

                            <div onclick="app.setSchool(${JSON.stringify(s).replace(/"/g, '&quot;')})" class="cursor-pointer">

                                <p class="font-bold text-gray-800 text-sm">${s.SCHUL_NM}</p>

                                <p class="text-[10px] text-gray-400">${s.ORG_RDNMA}</p>

                            </div>

                            <button onclick="app.removeSchoolFavorite('${s.SD_SCHUL_CODE}')" class="text-red-300"><i class="fa-solid fa-trash-can"></i></button>

                        </div>

                    `).join('') : `<p class="text-xs text-gray-400 italic">등록된 학교가 없습니다.</p>`}

                </div>`;


                list.innerHTML = html;

            }

        };


        const app = {

            async searchSchool() {

                const query = document.getElementById('school-search-input').value.trim();

                if (query.length < 2) return;

                ui.toggleLoader(true);

                try {

                    const res = await fetch(`https://open.neis.go.kr/hub/schoolInfo?Type=json&SCHUL_NM=${encodeURIComponent(query)}`);

                    const data = await res.json();

                    const container = document.getElementById('school-search-results');

                    container.innerHTML = '';

                    if (data.schoolInfo) {

                        data.schoolInfo[1].row.forEach(school => {

                            const div = document.createElement('div');

                            div.className = "bg-white p-4 rounded-2xl border border-gray-100 shadow-sm active:scale-95 transition-all cursor-pointer";

                            div.innerHTML = `<p class="font-bold text-gray-800">${school.SCHUL_NM}</p><p class="text-[10px] text-gray-400">${school.ORG_RDNMA}</p>`;

                            div.onclick = () => this.setSchool(school);

                            container.appendChild(div);

                        });

                    } else {

                        container.innerHTML = `<p class="text-center text-gray-400 py-6 text-sm">학교를 찾을 수 없습니다.</p>`;

                    }

                } catch (e) { console.error(e); }

                ui.toggleLoader(false);

            },


            setSchool(school) {

                state.currentSchool = school;

                localStorage.setItem('selectedSchool', JSON.stringify(school));

                ui.showSection('meal');

            },


            initMealView() {

                document.getElementById('display-school-name').textContent = state.currentSchool.SCHUL_NM;

                document.getElementById('display-school-location').textContent = state.currentSchool.ATPT_OFCDC_SC_NM;

                const picker = document.getElementById('meal-date-picker');

                picker.value = `${state.selectedDate.substring(0,4)}-${state.selectedDate.substring(4,6)}-${state.selectedDate.substring(6,8)}`;

                

                const isFav = state.favorites.some(f => f.SD_SCHUL_CODE === state.currentSchool.SD_SCHUL_CODE);

                document.getElementById('fav-toggle-btn').innerHTML = isFav ? '<i class="fa-solid fa-star text-yellow-400"></i>' : '<i class="fa-regular fa-star text-white"></i>';

                

                this.fetchMeal();

            },


            async fetchMeal() {

                ui.toggleLoader(true);

                const container = document.getElementById('meal-content');

                try {

                    const res = await fetch(`https://open.neis.go.kr/hub/mealServiceDietInfo?Type=json&ATPT_OFCDC_SC_CODE=${state.currentSchool.ATPT_OFCDC_SC_CODE}&SD_SCHUL_CODE=${state.currentSchool.SD_SCHUL_CODE}&MLSV_YMD=${state.selectedDate}`);

                    const data = await res.json();

                    container.innerHTML = '';

                    

                    if (data.mealServiceDietInfo) {

                        data.mealServiceDietInfo[1].row.forEach(meal => {

                            const menuItems = meal.DDISH_NM.split('<br/>').map(item => {

                                const allergyMatch = item.match(/\(([^)]+)\)/);

                                return { 

                                    name: item.replace(/\(([^)]+)\)/, '').trim(), 

                                    nums: allergyMatch ? allergyMatch[1].split('.') : [] 

                                };

                            });


                            const hasFav = menuItems.some(i => state.favDishes.includes(i.name));

                            const hasBad = menuItems.some(i => state.badDishes.includes(i.name));


                            const card = document.createElement('div');

                            card.className = `rounded-2xl p-5 shadow-sm border transition-all ${hasFav ? 'bg-blue-50 border-blue-200' : hasBad ? 'bg-red-50 border-red-100' : 'bg-white border-gray-100'}`;

                            card.innerHTML = `

                                <div class="flex justify-between items-start mb-4">

                                    <div>

                                        <span class="${hasFav ? 'bg-blue-500' : hasBad ? 'bg-red-500' : 'bg-gray-700'} text-white px-3 py-1 rounded-full text-[10px] font-bold uppercase">${meal.MMEAL_SC_NM}</span>

                                        ${hasFav ? '<p class="text-[10px] font-bold text-blue-500 mt-1">✨ 최애 반찬 포함!</p>' : ''}

                                        ${hasBad ? '<p class="text-[10px] font-bold text-red-500 mt-1">⚠️ 기피 반찬 주의</p>' : ''}

                                    </div>

                                    <span class="text-[10px] text-gray-400">${meal.CAL_INFO}</span>

                                </div>

                                <ul class="space-y-3">

                                    ${menuItems.map(item => {

                                        const isFav = state.favDishes.includes(item.name);

                                        const isBad = state.badDishes.includes(item.name);

                                        return `

                                        <li class="flex items-center justify-between">

                                            <div class="flex-1">

                                                <span class="text-sm font-medium ${isFav ? 'text-blue-700 font-bold' : isBad ? 'text-red-400 line-through' : 'text-gray-700'}">${item.name}</span>

                                                <span class="inline-block">${item.nums.map(n => `<span class="allergy-badge">${n}</span>`).join('')}</span>

                                            </div>

                                            <div class="flex gap-2">

                                                <button onclick="app.toggleFavDish('${item.name}')" class="p-1"><i class="${isFav ? 'fa-solid' : 'fa-regular'} fa-star ${isFav ? 'text-yellow-400' : 'text-gray-200'}"></i></button>

                                                <button onclick="app.toggleBadDish('${item.name}')" class="p-1"><i class="${isBad ? 'fa-solid' : 'fa-circle-minus'} ${isBad ? 'text-red-400' : 'text-gray-200'}"></i></button>

                                            </div>

                                        </li>`;

                                    }).join('')}

                                </ul>

                                <div class="mt-4 pt-4 border-t border-gray-100">

                                    <p class="text-[9px] text-gray-400 font-bold mb-1 uppercase tracking-tighter">이 식단의 유발 물질</p>

                                    <div class="flex flex-wrap gap-1">

                                        ${[...new Set(menuItems.flatMap(i => i.nums))].sort((a,b)=>a-b).map(n => `<span class="text-[9px] text-gray-500 bg-gray-100 px-1.5 py-0.5 rounded">${n}.${allergyMap[n]}</span>`).join('') || '<span class="text-[9px] text-gray-300">정보 없음</span>'}

                                    </div>
